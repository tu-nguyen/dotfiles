# twork ref
function tworktt() {
    printf "${BOLD}${H}twork cheatsheet${NC}\n"
    printf "%s\n" "$br"
    printf "${H2}cd${NC}\n"
    printf "  cd0 or cd-example1\n"
    printf "  cd1 or cd-example2\n"
    printf "  cd2 or cd-example3\n"
    printf "  cd3 or cd-example4\n"
    printf "%s\n" "$br"
    printf "${H2}ssh${NC}\n"
    printf "  ssh-example1 (example server 1)\n"
    printf "  ssh-example2 (example server 2)\n"
    printf "  ssh-example3 (example server 3)\n"
    printf "  ssh-example4 (example server 4)\n"
    printf "%s\n" "$br"
    printf "${H2}twork functions${NC}\n"

    # printf "  azr                  → Clear the Azure CLI cache\n"
    # printf "  print_curr_iter      → Print current Azure DevOps iter..\n"
    printf "%s\n" "$br"
 }

twork_motd() {
    printf "${BOLD}${H}Welcome to twork!${NC}\n"
    # printf "Run '${BOLD}${H}td${NC}' to display tasks.\n"
}

function td() {
    if [ -z "$FORCE_YES" ]; then
        read -p "${PINK}Display Tasks?${NC} [y/N] " -n 1 -r
        if [[ ! $REPLY =~ ^[Yy]$ ]]; then
            RETURN=1
            return
        else
            print_curr_iter "RESOURCE_GROUP" "NAME" "TEAM" "SECRET"
        fi
    fi
}
